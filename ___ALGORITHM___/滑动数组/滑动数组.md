1. 一般滑动数组，限制总数，不断右挪

``` csharp
//643
public double FindMaxAverage(int[] nums, int k) {
    int max = 0;
    for(int i = 0; i < k; i++)
    {
        max += nums[i];
    }

    int cur = max;
    for(int i = k; i < nums.Length; i++)
    {
        cur += nums[i];
        cur -= nums[i - k];
        max = Math.Max(max, cur);
    }

    return (double)max / k;
}
```

2. 限制某元素数量，超过时根据要求移动左面

``` csharp
// 1004
public int LongestOnes(int[] nums, int k) 
{
    int left = 0;
    int zeroCount = 0;
    int max = 0;

    for(int right = 0; right < nums.Length; right++)
    {
        if(nums[right] == 0)
        {
            zeroCount++;
        }

        while(zeroCount > k)
        {
            if(nums[left] == 0)
            {
                zeroCount--;
            }
            left++;
        }

        max = Math.Max(max, right - left + 1);
    }

    return max;
}
```
